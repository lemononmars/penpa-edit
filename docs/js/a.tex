  var2="${variants[i+1]}"

  # Handle odd numbered lists gracefully
  if [ -z "$var2" ]; then
    prompt="Implement the $var1 variant into the Sudoku editor frontend framework."
    session_name="install-$var1"
  else
    prompt="Implement both the $var1 and $var2 variants into the Sudoku editor frontend framework."
    session_name="install-$var1-and-$var2"
  fi

  echo "Launching Jules instance for: $session_name"
  
  # 3. Trigger the isolated concurrent task
  jules remote new --repo . --session "$session_name" --instruction "$prompt"
done
